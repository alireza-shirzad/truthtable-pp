\section{PIOPs for Strings}


\subsubsection{Activator Consistency Check}
\label{sec:act-consistency}
The \emph{Activator Consistency Check} takes the source column $\mathsf{src}$, the data activator $\mathsf{a_d}$, the length column $\mathsf{l}$, and the string activator $\mathsf{a_h}$, and proves that $\mathsf{a_d}$ and $\mathsf{a_h}$ are consistent: for every string, the number of its active characters equals its length when the string is active, and is $0$ otherwise. Equivalently, for every string index $i$,
\begin{displaymath}
  \mathsf{a_h}[i]\cdot\mathsf{l}[i] \;=\; \sum_{j:\,\mathsf{src}[j]=i} \mathsf{a_d}[j].
\end{displaymath}
We package this as the relation
\ifsp
\begin{equation*}
\NPRelation_{\mathsf{ActCons}} = \left\{
\begin{aligned}
  & \NPInstance = \bot,\; \\
  & \NPInstancey = (\Oracle{\MLE{\mathsf{src}}}, \Oracle{\MLE{\mathsf{a_d}}}, \Oracle{\MLE{\mathsf{l}}}, \Oracle{\MLE{\mathsf{a_h}}}),\; \\
  & \NPWitness = (\mathsf{src}, \mathsf{a_d}, \mathsf{l}, \mathsf{a_h}) \\
  & \;\big|\;\; \forall i,\; \mathsf{a_h}[i]\cdot\mathsf{l}[i] = \!\!\!\sum_{j:\,\mathsf{src}[j]=i}\!\!\! \mathsf{a_d}[j]
\end{aligned}
\right\}.
\end{equation*}
\else
\begin{align*}
\NPRelation_{\mathsf{ActCons}} = \left\{\,
  \NPInstance = \bot,\;
  \NPInstancey = (\Oracle{\MLE{\mathsf{src}}}, \Oracle{\MLE{\mathsf{a_d}}}, \Oracle{\MLE{\mathsf{l}}}, \Oracle{\MLE{\mathsf{a_h}}}),\;
  \NPWitness = (\mathsf{src}, \mathsf{a_d}, \mathsf{l}, \mathsf{a_h})
  \;\middle|\;
  \forall i,\; \mathsf{a_h}[i]\cdot\mathsf{l}[i] = \!\!\!\sum_{j:\,\mathsf{src}[j]=i}\!\!\! \mathsf{a_d}[j]
\,\right\}.
\end{align*}
\fi
The check is exactly a group-by-sum: grouping the rows $(\mathsf{src}, \mathsf{a_d})$ by $\mathsf{src}$ and summing $\mathsf{a_d}$ within each group must reproduce $\mathsf{l}$ on the active strings (those marked by $\mathsf{a_h}$). Treating $(\mathsf{src}, \mathsf{a_d})$ as an input table grouped by its key and $(\mathsf{ind}, \mathsf{l})$ with activator $\mathsf{a_h}$ as the grouped output, this is discharged by a single \cref{piop:keyedsumcheck}.
\begin{DefinePIOP}{Activator Consistency Check}{act-cons}
  \emph{Inputs:} $\Oracle{\MLE{\mathsf{src}}}, \Oracle{\MLE{\mathsf{a_d}}}, \Oracle{\MLE{\mathsf{l}}}, \Oracle{\MLE{\mathsf{a_h}}}$.
  \begin{enumerate}[nosep]
    \item $\PIOPProver$ and $\PIOPVerifier$ locally form the input table $\Left$ with key column $\Left.\key = \mathsf{src}$, value column $\Left.\val = \mathbf{1}$, and activator $\Left.\act = \mathsf{a_d}$ (so the keyed sum counts the active characters of each string).
    \item $\PIOPProver$ and $\PIOPVerifier$ locally form the output table $\Right$ with key column $\Right.\key = \mathsf{ind}$ (the index column, $\mathsf{ind}[i] = i$), value column $\Right.\val = \mathsf{l}$, and activator $\Right.\act = \mathsf{a_h}$.
    \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:keyedsumcheck} on $\Left$ and $\Right$, which forces $\sum_{j:\,\mathsf{src}[j]=i}\mathsf{a_d}[j] = \mathsf{a_h}[i]\cdot\mathsf{l}[i]$ for every key $i$.
  \end{enumerate}
\end{DefinePIOP}

\subsubsection{Length Filtering}
\label{sec:length-filtering}
The \emph{Length Filtering Check} takes the source column $\mathsf{src}$, the length column $\mathsf{l}$, the string activator $\mathsf{a_h^{old}}$, and the data activator $\mathsf{a_d^{old}}$ of a string column, together with a threshold $k$, and produces masked activators $\mathsf{a_h^{old'}}$ and $\mathsf{a_d^{old'}}$ that retain exactly the \emph{active} strings of length at least $k$ (and their characters), switching off every string shorter than $k$. We package this as the relation
\ifsp
\begin{equation*}
\NPRelation_{\mathsf{LenFilter}} = \left\{
\begin{aligned}
  & \NPInstance = k,\; \\
  & \NPInstancey = (\Oracle{\MLE{\mathsf{src}}}, \Oracle{\MLE{\mathsf{l}}}, \Oracle{\MLE{\mathsf{a_h^{old}}}}, \\
  & \qquad\qquad \Oracle{\MLE{\mathsf{a_d^{old}}}}, \Oracle{\MLE{\mathsf{a_h^{old'}}}}, \Oracle{\MLE{\mathsf{a_d^{old'}}}}),\; \\
  & \NPWitness = (\mathsf{src}, \mathsf{l}, \mathsf{a_h^{old}}, \mathsf{a_d^{old}}, \mathsf{a_h^{old'}}, \mathsf{a_d^{old'}}) \\
  & \;\big|\;\; \forall i:\, \mathsf{a_h^{old'}}[i] = \mathsf{a_h^{old}}[i]\cdot\mathbbm{1}[\mathsf{l}[i]\geq k], \\
  & \phantom{\;\big|\;\;} \forall j:\, \mathsf{a_d^{old'}}[j] = \mathsf{a_d^{old}}[j]\cdot\mathbbm{1}[\mathsf{l}[\mathsf{src}[j]]\geq k]
\end{aligned}
\right\}.
\end{equation*}
\else
\begin{equation*}
\NPRelation_{\mathsf{LenFilter}} = \left\{
\begin{aligned}
  & \NPInstance = k,\;
    \NPInstancey = (\Oracle{\MLE{\mathsf{src}}}, \Oracle{\MLE{\mathsf{l}}}, \Oracle{\MLE{\mathsf{a_h^{old}}}}, \Oracle{\MLE{\mathsf{a_d^{old}}}}, \Oracle{\MLE{\mathsf{a_h^{old'}}}}, \Oracle{\MLE{\mathsf{a_d^{old'}}}}),\; \\
  & \NPWitness = (\mathsf{src}, \mathsf{l}, \mathsf{a_h^{old}}, \mathsf{a_d^{old}}, \mathsf{a_h^{old'}}, \mathsf{a_d^{old'}}) \\
  & \;\big|\;\; \forall i:\, \mathsf{a_h^{old'}}[i] = \mathsf{a_h^{old}}[i]\cdot\mathbbm{1}[\mathsf{l}[i]\geq k]
    \;\text{ and }\; \forall j:\, \mathsf{a_d^{old'}}[j] = \mathsf{a_d^{old}}[j]\cdot\mathbbm{1}[\mathsf{l}[\mathsf{src}[j]]\geq k]
\end{aligned}
\right\}.
\end{equation*}
\fi
Removing short strings is essential to the prefix/suffix protocol: a string shorter than $k$ can never match a length-$k$ pattern, and once the anchored selectors are rotated inward, its selectors would run past its own end and spill into the character slots of its neighbors. The prover flags the short strings with two masks: $\mathsf{m_h}$, set to $1$ on each string of length less than $k$, and $\mathsf{m_d}$, set to $1$ on every character of such a string. A \cref{piop:boolean-check} keeps both masks $\{0,1\}$-valued, and an \cref{piop:act-cons} on $(\mathsf{src}, \mathsf{m_d}, \mathsf{l}, \mathsf{m_h})$ ties them together, forcing a string to be flagged by $\mathsf{m_h}$ exactly when all of its characters are flagged by $\mathsf{m_d}$. Two sign checks then certify that $\mathsf{m_h}$ flags \emph{exactly} the short strings: a \emph{Negative} \cref{piop:sign-check} over the flagged rows forces every flagged string to satisfy $\mathsf{l}-k<0$, while a \emph{Non-Negative} \cref{piop:sign-check} over the unflagged rows forces every unflagged string to satisfy $\mathsf{l}-k\geq0$. The masked activators are then the originals with the flagged strings switched off.
\begin{DefinePIOP}{Length Filtering Check}{len-filter}
  \emph{Inputs:} $\Oracle{\MLE{\mathsf{src}}}, \Oracle{\MLE{\mathsf{l}}}, \Oracle{\MLE{\mathsf{a_h^{old}}}}, \Oracle{\MLE{\mathsf{a_d^{old}}}}$, threshold $k$.
  \begin{enumerate}[nosep]
    \item $\PIOPProver$ sends the masks $\mathsf{m_h}$ and $\mathsf{m_d}$.
    \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:boolean-check} on $\mathsf{m_h}$ and on $\mathsf{m_d}$.
    \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:act-cons} on $(\mathsf{src}, \mathsf{m_d}, \mathsf{l}, \mathsf{m_h})$.
    \item $\PIOPProver$ and $\PIOPVerifier$ invoke the Negative \cref{piop:sign-check} on the column $(\mathsf{m_h}, \mathsf{l}-k)$.
    \item $\PIOPProver$ and $\PIOPVerifier$ invoke the Non-Negative \cref{piop:sign-check} on the column $(1-\mathsf{m_h}, \mathsf{l}-k)$.
    \item $\PIOPProver$ and $\PIOPVerifier$ define the masked activators as the derived oracles
    \begin{align*}
      \mathsf{a_d^{old'}} := \mathsf{a_d^{old}}\cdot(1-\mathsf{m_d}), \qquad
      \mathsf{a_h^{old'}} := \mathsf{a_h^{old}}\cdot(1-\mathsf{m_h}).
    \end{align*}
  \end{enumerate}
\end{DefinePIOP}
\begin{proof}
Booleanity forces $\mathsf{m_h}, \mathsf{m_d} \in \{0,1\}$. The two sign checks force $\mathsf{m_h}[i]=1 \iff \mathsf{l}[i]<k$: the Negative check rules out $\mathsf{l}[i]\geq k$ wherever $\mathsf{m_h}[i]=1$, and the Non-Negative check rules out $\mathsf{l}[i]<k$ wherever $\mathsf{m_h}[i]=0$. The activator-consistency check forces $\sum_{j:\,\mathsf{src}[j]=i}\mathsf{m_d}[j] = \mathsf{m_h}[i]\cdot\mathsf{l}[i]$ for every string $i$; as string $i$ has exactly $\mathsf{l}[i]$ characters and $\mathsf{m_d}$ is boolean, this forces $\mathsf{m_d}[j]=\mathsf{m_h}[\mathsf{src}[j]]$ for every character $j$, i.e.\ a character is flagged exactly when its string is. Hence $\mathsf{m_h}[i]=\mathbbm{1}[\mathsf{l}[i]<k]$ and $\mathsf{m_d}[j]=\mathbbm{1}[\mathsf{l}[\mathsf{src}[j]]<k]$, so the derived activators satisfy $\mathsf{a_h^{old'}}[i]=\mathsf{a_h^{old}}[i]\cdot\mathbbm{1}[\mathsf{l}[i]\geq k]$ and $\mathsf{a_d^{old'}}[j]=\mathsf{a_d^{old}}[j]\cdot\mathbbm{1}[\mathsf{l}[\mathsf{src}[j]]\geq k]$, exactly $\NPRelation_{\mathsf{LenFilter}}$. Completeness is immediate: the honest masks $\mathsf{m_h}=\mathbbm{1}[\mathsf{l}<k]$ and its per-character expansion $\mathsf{m_d}$ pass booleanity, activator-consistency, and both sign checks.
\end{proof}


    \subsubsection{Prefix and Suffix Check} A prefix/suffix check keeps the strings in a column that begin or end with a fixed pattern $\mathsf{str}$ of length $k$: a \emph{prefix} filter $\mathsf{str}\%$ keeps strings that start with $\mathsf{str}$ (as in \texttt{LIKE 'str\%'}), and a \emph{suffix} filter $\%\mathsf{str}$ keeps strings that end with it (as in \texttt{LIKE '\%str'}).

    \parhead{What the protocol proves}The verifier is given a tuple $( \mathsf{h},\mathsf{b},\mathsf{l},\mathsf{a_h^{old}}, \mathsf{d}, \mathsf{src}, \mathsf{a_d^{old}}, \mathsf{s_b})$ and new activators $\mathsf{a_h^{new}}$ and $\mathsf{a_d^{new}}$. The prover convinces the verifier that the tuple $( \mathsf{h},\mathsf{b},\mathsf{l},\mathsf{a_h^{new}}, \mathsf{d}, \mathsf{src}, \mathsf{a_d^{new}}, \mathsf{s_b})$ is the result of applying the filter to the original tuple.


At a high level the check has several parts, which we explain in turn before presenting the protocol. For concreteness we describe the prefix scenario throughout; afterwards we show how the prefix and suffix protocols can be unified into a single check.


\parhead{Activator consistency} The prover claims that $\mathsf{a_h^{new}}$ and $\mathsf{a_d^{new}}$ are the activators of the filtered column. The verifier checks that they are internally consistent with one another and with the lengths: a string is marked active by $\mathsf{a_h^{new}}$ exactly when its characters are marked active by $\mathsf{a_d^{new}}$, and the active-character count of each string equals its length.

\parhead{Filtering by length} A string shorter than $k$ can never match a length-$k$ pattern, and leaving it in place is actively harmful: once the anchored selectors are rotated inward, a short string's selectors run past its own end and spill into the character slots of its neighbors, corrupting the comparison. The first step therefore removes every string of length less than $k$ by invoking the \cref{piop:len-filter} (\cref{sec:length-filtering}), which returns masked activators $\mathsf{a_h^{old'}}$ and $\mathsf{a_d^{old'}}$ equal to $\mathsf{a_h^{old}}, \mathsf{a_d^{old}}$ with every short string (and all of its characters) switched off. All later steps use these in place of the originals, so that they range only over strings long enough to be candidate matches.



\parhead{Containment} A string can only survive the filter if it was eligible to begin with. A \cref{piop:zerocheck} on $\mathsf{a_h^{new}}\cdot(1-\mathsf{a_h^{old'}})$ forces $\mathsf{a_h^{new}}\leq\mathsf{a_h^{old'}}$, so the prover cannot retain a short or inactive string --- one whose anchored slots, and hence the pattern comparison below, are switched off. Every kept string is therefore eligible, with all $k$ of its anchored slots live.

\parhead{False positives Check} A string that \emph{survives} the filter must genuinely match the pattern. The pattern column $\mathsf{p}$ already holds the required character at every anchored slot, so this reduces to a single check that the data agrees with $\mathsf{p}$ there. The parties run a \cref{piop:zerocheck} on $\mathsf{d}-\mathsf{p}$ with activator $\mathsf{a_d^{new}}\cdot\sum_{i=0}^{k-1}\mathsf{s'_b}^{(i)}$, which is on exactly at the anchored slots of the kept strings; the check therefore forces $\mathsf{d}=\mathsf{p}$ at every one of them, so no kept string disagrees with the pattern at any position.

\parhead{False negatives Check} Conversely, a string that the prover \emph{drops} must genuinely fail the pattern; otherwise a real match would be silently discarded. A single mismatched position is enough to rule a string out, so rather than re-prove the whole comparison, the prover exhibits one witness per dropped string. It sends a boolean \emph{mismatch selector} $\mathsf{s_n}$ that places a single $1$ inside each \emph{dropped candidate} --- a string that was eligible ($\mathsf{a_d^{old'}}=1$: active and at least $k$ long) but not kept ($\mathsf{a_d^{new}}=0$) --- at an anchored slot where its character violates the pattern, and is $0$ everywhere else. For this to rule out false negatives, $\mathsf{s_n}$ must be \emph{well-formed}: it must place exactly one mark in each non-matching eligible string, and that mark must sit at a genuine mismatch. We establish this with a counting argument. A \cref{piop:boolean-check} forces $\mathsf{s_n}$ to be $\{0,1\}$-valued, a \cref{piop:nodup-check} on the source indices carried by the marked slots forces those slots to lie in distinct strings, so each string holds at most one mark, and a \cref{piop:zerocheck} on $\mathsf{s_n}\cdot(1-\sum_{i=0}^{k-1}\mathsf{s'_b}^{(i)})$ confines every mark to an anchored slot of an eligible string (so a mark cannot be parked on an inactive slot where $\mathsf{p}=0$ would make the mismatch test pass vacuously). The prover then announces the number of matches $n_{\mathsf{m}}$ and non-matches $n_{\mathsf{nm}}$, and three sumchecks pin these down: $\sum\mathsf{a_h^{new}} = n_{\mathsf{m}}$ counts the kept strings, $\sum\mathsf{s_n} = n_{\mathsf{nm}}$ counts the marks, and $\sum\mathsf{a_h^{old'}} = n_{\mathsf{m}}+n_{\mathsf{nm}}$ counts the eligible strings. Since every eligible string is either kept or marked but not both (a marked slot has $\mathsf{d}\neq\mathsf{p}$, which a kept string never has), the identity $n_{\mathsf{m}}+n_{\mathsf{nm}}=\sum\mathsf{a_h^{old'}}$ forces the marks to cover \emph{every} non-matching eligible string. Finally, a \cref{piop:nozero-check} on $\mathsf{d}-\mathsf{p}$ with activator $\mathsf{s_n}$ confirms every marked slot is a genuine mismatch ($\mathsf{d}\neq\mathsf{p}$ there), so each dropped string really fails the pattern.

\parhead{Anchored selectors and the pattern column} Let $\rho_t(\cdot)$ denote rotation of a column by $t$ positions (a $1$ at index $p$ moves to index $p+t$). The boundary selector $\mathsf{s_b}$ from the tuple marks each string's first character. Once the masked activator $\mathsf{a_d^{old'}}$ is in hand (it switches off short and inactive strings; see below), the parties form the \emph{masked boundary selector}
\begin{displaymath}
  \mathsf{s'_b}^{(0)} := \mathsf{a_d^{old'}}\cdot\mathsf{s_b},
\end{displaymath}
which keeps a boundary mark only on the strings that survive into the comparison. The prover then sends the further \emph{anchored selectors} $\mathsf{s'_b}^{(1)},\ldots,\mathsf{s'_b}^{(k-1)}$, defined by
\begin{displaymath}
  \mathsf{s'_b}^{(i)} := \rho_{i}\bigl(\mathsf{s'_b}^{(0)}\bigr), \qquad i = 0,\ldots,k-1,
\end{displaymath}
so that $\mathsf{s'_b}^{(i)}$ marks the $(i{+}1)$-th character of every surviving string. Each $\mathsf{s'_b}^{(i)}$ is tied to $\mathsf{s'_b}^{(0)}$ by a \cref{piop:rotcheck}. Because the short strings were removed \emph{before} rotating, no selector runs off the end of a string into a neighbor, so the marks never collide. The parties then define the \emph{pattern column}
\begin{displaymath}
  \mathsf{p} := \sum_{i=0}^{k-1} \mathsf{s'_b}^{(i)}\cdot\mathsf{str}[i],
\end{displaymath}
which carries the required pattern character $\mathsf{str}[i]$ at the $(i{+}1)$-th slot of every surviving string and is $0$ elsewhere --- exactly the pattern the prefix filter expects laid over the anchored slots.

\emph{Unifying prefix and suffix.} The two cases differ only in which end the comparison is anchored to. A suffix filter anchors at each string's \emph{last} character, so the masked anchor is $\mathsf{s'_a}^{(0)} := \mathsf{a_d^{old'}}\cdot\rho_{-1}(\mathsf{s_b})$ and the prover sends the backward rotations $\mathsf{s'_a}^{(i)} := \rho_{-i}\bigl(\mathsf{s'_a}^{(0)}\bigr)$, forming the pattern column $\mathsf{p} := \sum_{i=0}^{k-1}\mathsf{s'_a}^{(i)}\cdot\mathsf{str}[k-1-i]$, which places $\mathsf{str}[k-1]$ on the last character, $\mathsf{str}[k-2]$ on the second-to-last, and so on. Writing $\epsilon := +1$ for a prefix and $\epsilon := -1$ for a suffix, both cases use masked anchored selectors $\rho_{\epsilon i}(\text{anchor})$ and pattern characters $\mathsf{str}[\pi(i)]$ with $\pi(i) := \tfrac{(k-1)(1-\epsilon)}{2} + \epsilon\, i$, so $\pi(i)=i$ for a prefix and $k-1-i$ for a suffix.


   \begin{DefinePIOP}{Prefix/Suffix Check}{prefix-check}
    \begin{enumerate}[nosep]
        \item (\textit{Activator validity check})
        \begin{enumerate}[nosep]
          \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:boolean-check} on $\mathsf{a_d^{new}}$ and on $\mathsf{a_h^{new}}$.
          \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:act-cons} on $(\mathsf{src}, \mathsf{a_d^{new}}, \mathsf{l}, \mathsf{a_h^{new}})$.
        \end{enumerate}
        \item (\textit{Filtering by length}) $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:len-filter} on $(\mathsf{src}, \mathsf{l}, \mathsf{a_h^{old}}, \mathsf{a_d^{old}})$ with threshold $k$, obtaining the masked activators $\mathsf{a_h^{old'}}$ and $\mathsf{a_d^{old'}}$.
        \item (\textit{Containment}) $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:zerocheck} on the column $\mathsf{a_h^{new}}\cdot(1-\mathsf{a_h^{old'}})$, forcing every kept string to be eligible.
        \item (\textit{Pattern selectors}) $\PIOPProver$ and $\PIOPVerifier$ define $\mathsf{s'_b}^{(0)} := \mathsf{a_d^{old'}}\cdot\mathsf{s_b}$ as the first pattern selector, then
       \begin{enumerate}[nosep]
        \item $\PIOPProver$ sends $\mathsf{s'_b}^{(1)},\ldots,\mathsf{s'_b}^{(k-1)}$, each $\mathsf{s'_b}^{(i)}$ being the rotation of $\mathsf{s'_b}^{(i-1)}$ by $\epsilon$.
        \item For $i=1,\ldots,k-1$ the parties invoke \cref{piop:rotcheck} on $(\mathsf{s'_b}^{(i-1)}, \mathsf{s'_b}^{(i)}, \epsilon)$. 
        \item $\PIOPProver$ and $\PIOPVerifier$ then define the pattern column $\mathsf{p} := \sum_{i=0}^{k-1}\mathsf{s'_b}^{(i)}\cdot\mathsf{str}[\pi(i)]$.
       \end{enumerate} 
        \item (\textit{False Positive Check}) $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:zerocheck} on the column $\bigl(\mathsf{d}-\mathsf{p},\;\;\mathsf{a_d^{new}}\cdot\textstyle\sum_{i=0}^{k-1}\mathsf{s'_b}^{(i)}\bigr)$.
        \item (\textit{False Negative Check}) $\PIOPProver$ sends the boolean mismatch selector $\mathsf{s_n}$.
        \begin{enumerate}[nosep]
          \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:nozero-check} on the column $(\mathsf{d}-\mathsf{p},\;\mathsf{s_n})$, confirming each marked slot is a genuine mismatch.
          \item (\textit{$\mathsf{s_n}$ well-formedness})
          \begin{enumerate}[nosep]
            \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:boolean-check} on $\mathsf{s_n}$.
            \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:nodup-check} on the column with data $\mathsf{src}$ and activator $\mathsf{s_n}$, so no two marks fall in the same string.
            \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:zerocheck} on the column $\mathsf{s_n}\cdot\bigl(1-\sum_{i=0}^{k-1}\mathsf{s'_b}^{(i)}\bigr)$, confining each mark to an anchored slot of an eligible string.
            \item $\PIOPProver$ sends the match count $n_{\mathsf{m}}$ and the non-match count $n_{\mathsf{nm}}$.
            \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:sumcheck} on $\mathsf{a_h^{new}}$ with claimed sum $n_{\mathsf{m}}$ (the number of matches).
            \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:sumcheck} on $\mathsf{s_n}$ with claimed sum $n_{\mathsf{nm}}$ (the number of non-matches).
            \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:sumcheck} on $\mathsf{a_h^{old'}}$ with claimed sum $n_{\mathsf{m}}+n_{\mathsf{nm}}$ (the number of eligible strings).
          \end{enumerate}
        \end{enumerate}
    \end{enumerate}
\end{DefinePIOP}

\begin{proof}
We argue completeness and soundness, assuming each invoked sub-PIOP is complete and sound for its relation. Call a string \emph{eligible} if it is active under $\mathsf{a_h^{old}}$ and has length at least $k$; by construction these are exactly the strings marked by $\mathsf{a_h^{old'}}$, and we write $E := \{\,i : \mathsf{a_h^{old'}}[i]=1\,\}$ for this set. The filter must keep exactly the eligible strings that match $\mathsf{str}$.

\parhead{Completeness}
By the completeness of \cref{piop:len-filter}, an honest prover's masked activators $\mathsf{a_h^{old'}},\mathsf{a_d^{old'}}$ switch off precisely the short and inactive strings. The masked boundary selector and its rotations are genuine rotations, so the rotation checks pass and $\mathsf{p}$ carries $\mathsf{str}[\pi(i)]$ on the $(i{+}1)$-th anchored slot of every eligible string. The honest new activators keep exactly the eligible matching strings, so they are boolean, consistent, contained in $\mathsf{a_h^{old'}}$, and agree with $\mathsf{p}$ on every anchored slot of a kept string, passing the activator-validity and false-positive checks. For each eligible non-matching string the prover marks one anchored slot where $\mathsf{d}\neq\mathsf{p}$; this $\mathsf{s_n}$ is boolean, places no two marks in one string, lies on anchored slots, and the announced counts satisfy the three sum checks, so every false-negative check passes. Hence the honest transcript is accepted.

\parhead{Soundness}
Suppose $\PIOPVerifier$ accepts. We show that the kept set $M := \{\,i : \mathsf{a_h^{new}}[i]=1\,\}$ is exactly the set of eligible matching strings; since a match requires length at least $k$ and original activity, this is precisely the set of strings passing the filter.

By the soundness of \cref{piop:len-filter} (\cref{sec:length-filtering}), the masked activators $\mathsf{a_h^{old'}}, \mathsf{a_d^{old'}}$ are on exactly for the active strings of length at least $k$ and their characters; hence $E = \{\,i : \mathsf{a_h^{old'}}[i]=1\,\}$ is precisely the set of eligible strings, as claimed.

The activator-validity check makes $\mathsf{a_h^{new}},\mathsf{a_d^{new}}$ boolean and mutually consistent: a string is on in $\mathsf{a_h^{new}}$ iff all of its characters are on in $\mathsf{a_d^{new}}$, and its active-character count equals its length. The containment zerocheck $\mathsf{a_h^{new}}(1-\mathsf{a_h^{old'}})=0$ forces $M\subseteq E$, so every kept string is eligible and, by consistency, all of its characters carry $\mathsf{a_d^{new}}=1$.

\emph{No false positives.} For $i\in M$ the anchored slots of string $i$ have $\mathsf{a_d^{new}}=1$ and $\sum_t \mathsf{s'_b}^{(t)}=1$, so the false-positive zerocheck forces $\mathsf{d}=\mathsf{p}$ there. As $\mathsf{p}$ carries $\mathsf{str}[\pi(t)]$ at the $(t{+}1)$-th anchored slot, string $i$ matches $\mathsf{str}$. Thus every kept string is an eligible match.

\emph{No false negatives.} Let $N := \{\,i : \sum_{\mathsf{src}[j]=i}\mathsf{s_n}[j]=1\,\}$ be the marked strings. Booleanity of $\mathsf{s_n}$ and the no-duplicate check on $(\mathsf{src},\mathsf{s_n})$ give at most one mark per string, so $\sum\mathsf{s_n}=|N|$, while the confinement zerocheck $\mathsf{s_n}(1-\sum_t\mathsf{s'_b}^{(t)})=0$ places every mark on an anchored slot of an eligible string, so $N\subseteq E$. The nonzero check on $(\mathsf{d}-\mathsf{p},\mathsf{s_n})$ makes $\mathsf{d}\neq\mathsf{p}$ at each marked slot; since that slot is anchored, $\mathsf{p}=\mathsf{str}[\pi(t)]$, so the marked string mismatches $\mathsf{str}$ and is non-matching. In particular $N\cap M=\emptyset$, since a kept eligible string agrees with $\mathsf{p}$ on all of its anchored slots.

The three sum checks give $|M|=\sum\mathsf{a_h^{new}}=n_{\mathsf{m}}$, $|N|=\sum\mathsf{s_n}=n_{\mathsf{nm}}$, and $|E|=\sum\mathsf{a_h^{old'}}=n_{\mathsf{m}}+n_{\mathsf{nm}}=|M|+|N|$. As $M,N\subseteq E$ are disjoint with sizes summing to $|E|$, they partition $E=M\sqcup N$. Every eligible string is therefore either kept or marked, and the marked ones are non-matching, so no eligible matching string is dropped. Combined with the no-false-positive direction, $M$ is exactly the set of eligible matching strings, i.e.\ the strings passing the prefix filter. The suffix case is identical, with $\epsilon=-1$ and $\pi(i)=k-1-i$.
\end{proof}

\subsubsection{Infix Check}
\label{sec:infix-check}
An \emph{infix} (substring) filter $\%\mathsf{str}\%$ keeps the strings of a column that contain a fixed pattern $\mathsf{str}$ of length $k$ as a contiguous substring at \emph{any} position, as in \texttt{LIKE '\%str\%'}. Unlike the prefix and suffix filters, the pattern is not anchored to either end: a string of length $l$ has $l-k+1$ candidate windows, and it matches if \emph{any} one of them equals $\mathsf{str}$.

The starting point is to compare every length-$k$ window against $\mathsf{str}$ at once. The prover first lays down the rotated character columns needed to read each window in parallel: writing $\mathsf{d}^{(0)} := \mathsf{d}$, the column $\mathsf{d}^{(\delta)}$ holds, at each position $p$, the character $\mathsf{d}[p+\delta]$, so that the tuple $(\mathsf{d}^{(0)}[p],\ldots,\mathsf{d}^{(k-1)}[p])$ spells out the length-$k$ window of $\mathsf{d}$ beginning at $p$. The verifier then collapses each window into a single field element by a random linear combination: it sends coefficients $r_0,\ldots,r_{k-1}$, and the parties form the \emph{window fingerprint} $\mathsf{wf}[p] := \sum_{\delta=0}^{k-1} r_\delta\,\mathsf{d}^{(\delta)}[p]$ together with the scalar \emph{pattern fingerprint} $\mathsf{pf} := \sum_{\delta=0}^{k-1} r_\delta\,\mathsf{str}[\delta]$. A window that equals $\mathsf{str}$ has $\mathsf{wf}[p]=\mathsf{pf}$ for every choice of coefficients, whereas a window that differs has $\mathsf{wf}[p]=\mathsf{pf}$ only with probability $(k-1)/|\F|$ over the verifier's coins. The length-$k$ comparison at every window is thereby reduced to the single field condition $\mathsf{wf}[p]=\mathsf{pf}$. Equivalently, the parties form the \emph{difference column} $\mathsf{diff} := \mathsf{wf}-\mathsf{pf}$, which vanishes exactly at the starting positions of the occurrences of $\mathsf{str}$.

\parhead{Window validity} Not every position starts a genuine window: the last $k-1$ characters of a string have fewer than $k$ characters left, so their windows run off the end (or into the next string). The parties single these out with the boundary selector $\mathsf{s_b}$, which marks each string's first character: rotating it left by $1,\ldots,k-1$ yields columns $\mathsf{b}^{(j)} := \rho_{-j}(\mathsf{s_b})$ marking, respectively, the last, second-to-last, \ldots, $(k-1)$-th-from-last character of every string. Subtracting these from the masked activator gives the \emph{window-validity mask} $\mathsf{v} := \mathsf{a_d^{old'}}\cdot(1 - \sum_{j=1}^{k-1}\mathsf{b}^{(j)})$, which is $1$ exactly on the valid window starts --- the first $\mathsf{l}-k+1$ characters --- of each eligible string and $0$ elsewhere.

\parhead{False negatives} A string that the filter \emph{drops} must genuinely avoid the pattern: every one of its windows must fail, where a window fails exactly when $\mathsf{diff}[p]\neq0$. The prover exhibits the failure by sending a Boolean false-negative witness $\mathsf{w_n}$ that marks one valid window inside each dropped eligible string. A \cref{piop:boolean-check} keeps $\mathsf{w_n}$ Boolean, and a \cref{piop:nozero-check} on $\mathsf{diff}$ with activator $\mathsf{w_n}\cdot\mathsf{v}\cdot(1-\mathsf{a_d^{new}})\cdot\mathsf{a_d^{old}}$ certifies that every witnessed window genuinely fails ($\mathsf{diff}\neq0$). The confining factors restrict each mark to a valid window ($\mathsf{v}$, which already excludes the inactive and short strings as well as the trailing $k-1$ positions where a window would run off the end) of an active ($\mathsf{a_d^{old}}$) string the prover did \emph{not} keep ($1-\mathsf{a_d^{new}}$). Thus a mark cannot be parked on an inactive string, a degenerate boundary-crossing window, or a kept string in order to dodge the test.

\parhead{False positives} Conversely, a string that the filter \emph{keeps} must genuinely contain the pattern: it must have at least one window with $\mathsf{diff}=0$. The prover exhibits such a window by sending a Boolean \emph{witness selector} $\mathsf{w}$ that marks, inside each kept string, a matching position. A \cref{piop:boolean-check} keeps $\mathsf{w}$ Boolean, and a \cref{piop:zerocheck} on the column with data $\mathsf{diff}$ and activator $\mathsf{w}\cdot\mathsf{a_d^{new}}$ asserts that every witnessed position of a kept string is a genuine occurrence of $\mathsf{str}$ ($\mathsf{diff}=0$ there); the activator $\mathsf{a_d^{new}}$ confines attention to the kept strings, so marks outside them are ignored. It remains to force \emph{every} kept string to carry such a mark, ruling out a prover that keeps a non-containing string under a vacuous $\mathsf{w}\equiv0$. This is a counting argument: the prover announces the number of kept strings $n_{\mathsf{m}}$, and two \cref{piop:sumcheck} invocations pin down $\sum\mathsf{w}=n_{\mathsf{m}}$ and $\sum\mathsf{a_h^{new}}=n_{\mathsf{m}}$, so $\mathsf{w}$ holds exactly $n_{\mathsf{m}}$ marks; a \cref{piop:nodup-check} on the source indices $\mathsf{src}$ carried by the marked positions, with activator $\mathsf{w}$, forces those marks to lie in distinct strings, so each kept string holds at most one. With $n_{\mathsf{m}}$ marks spread over the $n_{\mathsf{m}}$ kept strings, no two in the same string, every kept string carries exactly one witnessed occurrence of $\mathsf{str}$ --- so each one genuinely contains the pattern.

\begin{DefinePIOP}{Infix Check}{infix-check}
  \begin{enumerate}[nosep]
    \item (\textit{Rotated character columns}) Writing $\mathsf{d}^{(0)} := \mathsf{d}$, $\PIOPProver$ sends the rotated character columns $\mathsf{d}^{(1)},\ldots,\mathsf{d}^{(k-1)}$, each $\mathsf{d}^{(\delta)}$ being the rotation of $\mathsf{d}^{(\delta-1)}$ by $-1$, so that $\mathsf{d}^{(\delta)}[p] = \mathsf{d}[p+\delta]$.
    \begin{enumerate}[nosep]
      \item For $\delta = 1,\ldots,k-1$, $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:rotcheck} on $(\mathsf{d}^{(\delta-1)}, \mathsf{d}^{(\delta)}, -1)$.
    \end{enumerate}
    \item (\textit{Window fingerprint}) $\PIOPVerifier$ samples $k$ random field elements $r_0,\ldots,r_{k-1} \samples \F$ and sends them to $\PIOPProver$. The parties form the window-fingerprint column $\mathsf{wf} := \sum_{\delta=0}^{k-1} r_\delta\,\mathsf{d}^{(\delta)}$ and the pattern fingerprint $\mathsf{pf} := \sum_{\delta=0}^{k-1} r_\delta\,\mathsf{str}[\delta]$, so that $\mathsf{wf}[p] = \mathsf{pf}$ exactly when the length-$k$ window of $\mathsf{d}$ at $p$ agrees with $\mathsf{str}$ on the verifier's combination.
    \item (\textit{Difference column}) The parties form $\mathsf{diff} := \mathsf{wf} - \mathsf{pf}$, which satisfies $\mathsf{diff}[p] = 0$ exactly at the positions $p$ where the length-$k$ window of $\mathsf{d}$ equals $\mathsf{str}$ (up to the $(k-1)/|\F|$ fingerprint-collision error), i.e.\ at the starting positions of the occurrences of $\mathsf{str}$.
    \item (\textit{Window-validity mask}) $\PIOPProver$ sends the rotated boundary columns $\mathsf{b}^{(1)},\ldots,\mathsf{b}^{(k-1)}$, where $\mathsf{b}^{(0)} := \mathsf{s_b}$ and each $\mathsf{b}^{(j)}$ is the rotation of $\mathsf{b}^{(j-1)}$ by $-1$, so that $\mathsf{b}^{(j)}$ marks the $j$-th character from the end of every string.
    \begin{enumerate}[nosep]
      \item For $j=1,\ldots,k-1$, $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:rotcheck} on $(\mathsf{b}^{(j-1)}, \mathsf{b}^{(j)}, -1)$.
      \item The parties define the window-validity mask $\mathsf{v} := \mathsf{a_d^{old'}}\cdot\bigl(1 - \textstyle\sum_{j=1}^{k-1}\mathsf{b}^{(j)}\bigr)$, which is $1$ exactly on the first $\mathsf{l}-k+1$ characters of each eligible string --- its valid window starts --- and $0$ on the last $k-1$.
    \end{enumerate}
    \item (\textit{False negative check}) $\PIOPProver$ sends a Boolean false-negative witness $\mathsf{w_n}$ that marks, inside each \emph{dropped} eligible string, one valid window that fails to match $\mathsf{str}$.
    \begin{enumerate}[nosep]
      \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:boolean-check} on $\mathsf{w_n}$.
      \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:nozero-check} on the column with data $\mathsf{diff}$ and activator $\mathsf{w_n}\cdot\mathsf{v}\cdot(1-\mathsf{a_d^{new}})\cdot\mathsf{a_d^{old}}$, certifying $\mathsf{diff}\neq0$ at every witnessed window --- so the witnessed window of each dropped string genuinely fails the pattern. The factors $\mathsf{v}\cdot(1-\mathsf{a_d^{new}})\cdot\mathsf{a_d^{old}}$ confine every mark to a valid window of an active, dropped string.
    \end{enumerate}
    \item (\textit{False positive check}) $\PIOPProver$ sends a witness selector $\mathsf{w}$ that marks, inside each kept string, a position where a window matches $\mathsf{str}$.
    \begin{enumerate}[nosep]
      \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:boolean-check} on $\mathsf{w}$.
      \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:zerocheck} on the column with data $\mathsf{diff}$ and activator $\mathsf{w}\cdot\mathsf{a_d^{new}}$, asserting $\mathsf{diff}=0$ at every witnessed position of a kept string --- so each witness is a genuine occurrence of $\mathsf{str}$.
      \item $\PIOPProver$ announces the number of kept strings $n_{\mathsf{m}}$, and the parties invoke \cref{piop:sumcheck} twice to check $\sum_{\vec{x}}\mathsf{w}(\vec{x}) = n_{\mathsf{m}}$ and $\sum_{\vec{x}}\mathsf{a_h^{new}}(\vec{x}) = n_{\mathsf{m}}$, forcing $\mathsf{w}$ to carry exactly as many marks as there are kept strings.
      \item $\PIOPProver$ and $\PIOPVerifier$ invoke \cref{piop:nodup-check} on the column with data $\mathsf{src}$ and activator $\mathsf{w}$, forcing the marked positions to lie in distinct strings --- so each kept string carries at most one witness. Together with the count match, every kept string carries \emph{exactly} one witnessed occurrence of $\mathsf{str}$.
    \end{enumerate}
  \end{enumerate}
\end{DefinePIOP}
